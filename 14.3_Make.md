
# Make

> 동시에 여러파일을 퍼블리시 하는 코드를 작성한다.

#### GNU make파일 작성법

	EXPORT_DIR = publish
	EPUB := $(patsubst %.md,%.epub,$(wildcard *.md))
	all: $(addprefix $(EXPORT_DIR)/, $(EPUB))
	$(EXPORT_DIR) :
		mkdir -p $@
	
	$(EXPORT_DIR)/%.epub : %.md $(EXPORT_DIR)
		pandoc $< -o $@ -s
	
	clean:
		rm -rf $(EXPORT_DIR)
